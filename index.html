// app.js - Complete Backend + Frontend in ONE file const express = require('express'); const axios = require('axios'); const cors = require('cors'); const app = express(); app.use(cors()); app.use(express.json()); const OUTLOOK_API = 'https://graph.microsoft.com/v1.0/me/messages'; // Extract OTP from text function extractOTP(text) { if (!text) return null; text = text.replace(/"/g, '"') .replace(/&/g, '&') .replace(/</g, '<') .replace(/>/g, '>') .replace(/&#(\d+);/g, (match, dec) => String.fromCharCode(dec)); const allCodes = []; const p1 = text.match(/(\d{4,8})\s+is\s+your/gi); if (p1) p1.forEach(m => { const c = m.match(/(\d{4,8})/); if(c) allCodes.push(c[1]); }); const p2 = text.match(/code[\s:is]+(\d{4,8})/gi); if (p2) p2.forEach(m => { const c = m.match(/(\d{4,8})/); if(c) allCodes.push(c[1]); }); const p3 = text.match(/\b(\d{4,8})\b/g); if (p3) { p3.forEach(code => { if (!['2024', '2025', '2026', '8080', '443', '80', '1234', '0000'].includes(code)) { allCodes.push(code); } }); } return allCodes.length > 0 ? allCodes[allCodes.length - 1] : null; } // API endpoint app.post('/api/get-code', async (req, res) => { try { const { email, password, token, uuid, service } = req.body; if (!token) return res.json({ success: false, error: 'Token required' }); console.log(`📧 Fetching ${service} code for ${email}...`); const response = await axios.get(OUTLOOK_API, { headers: { 'Authorization': `Bearer ${token}`, 'Content-Type': 'application/json' }, params: { '$top': 10, '$orderby': 'receivedDateTime DESC' } }); const messages = response.data.value || []; const keywords = service === 'instagram' ? ['instagram', 'ig'] : ['facebook', 'fb']; let targetEmail = messages.find(msg => { const from = (msg.from?.emailAddress?.address || '').toLowerCase(); const subject = (msg.subject || '').toLowerCase(); return keywords.some(k => from.includes(k) || subject.includes(k)); }); if (!targetEmail) return res.json({ success: false, error: `No ${service} email found` }); const fullText = `${targetEmail.subject} ${targetEmail.bodyPreview} ${targetEmail.body?.content}`; const code = extractOTP(fullText); if (code) { console.log(`✅ Code found: ${code}`); return res.json({ success: true, code: code, email: email, service: service }); } return res.json({ success: false, error: 'No code found in email' }); } catch (error) { return res.json({ success: false, error: error.message }); } }); // Serve HTML frontend with FIXED COLORS app.get('/', (req, res) => { res.send(` Email Verification Code Extractor

------

  • • Format: email|password|token|uuid
  • • Supports 5-8 digit FB/IG codes
    • `); }); app.listen(3000, () => console.log('✅ Server on http://localhost:3000'));
