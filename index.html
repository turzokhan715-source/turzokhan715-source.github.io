// app.js - Optimized OTP Extractor const express = require('express'); const axios = require('axios'); const cors = require('cors'); const app = express(); app.use(cors()); app.use(express.json()); // Outlook API URL const OUTLOOK_API = 'https://graph.microsoft.com/v1.0/me/messages'; // এই ফাংশনটি এখন ৫-৮ ডিজিটের কোড আরও গভীরভাবে খুঁজবে function extractOTP(text) { if (!text) return null; // ৫ থেকে ৮ ডিজিটের নম্বর খোঁজার নিয়ম const otpRegex = /\b\d{5,8}\b/g; const matches = text.match(otpRegex); if (matches) { // সাধারণত সবচেয়ে শেষের কোডটিই লেটেস্ট ওটিপি হয় return matches[matches.length - 1]; } return null; } // API endpoint app.post('/api/get-code', async (req, res) => { try { const { email, token, service } = req.body; if (!token) return res.json({ success: false, error: 'Token required' }); // আমরা সরাসরি ইনবক্সের শেষ ২০টি মেসেজ চেক করবো যেন মিস না হয় const response = await axios.get(OUTLOOK_API, { headers: { 'Authorization': `Bearer ${token}` }, params: { '$top': 20, '$orderby': 'receivedDateTime DESC', '$select': 'subject,bodyPreview,body' // বডিও চেক করবো } }); const messages = response.data.value || []; const keywords = service === 'instagram' ? ['instagram', 'ig'] : ['facebook', 'fb', 'security']; // কিউওয়ার্ড দিয়ে ইমেইল খোঁজা let targetMsg = messages.find(m => { const subject = (m.subject || '').toLowerCase(); const preview = (m.bodyPreview || '').toLowerCase(); return keywords.some(k => subject.includes(k) || preview.includes(k)); }); if (!targetMsg) return res.json({ success: false, error: 'Email not found in Inbox' }); // সাবজেক্ট, প্রিভিউ এবং মেইন বডি—সব জায়গায় কোড খুঁজবে const contentToSearch = `${targetMsg.subject} ${targetMsg.bodyPreview} ${targetMsg.body?.content || ''}`; const code = extractOTP(contentToSearch); if (code) { return res.json({ success: true, code: code }); } else { return res.json({ success: false, error: 'Code not found in the email content' }); } } catch (error) { console.error(error.message); return res.json({ success: false, error: 'Connection Error with Outlook' }); } }); // HTML Frontend (Fixed Colors & Design) app.get('/', (req, res) => { res.send(`
