Production-e use korar jonno tumake: Node.js/Express backend setup korte hobe Microsoft Graph API integration korte hobe Proper OAuth authentication implement korte hobe Backend Code (Node.js/Express): const express = require('express'); const axios = require('axios'); const app = express(); app.use(express.json()); app.post('/api/get-otp', async (req, res) => { const { email, token } = req.body; try { // Microsoft Graph API call const response = await axios.get( 'https://graph.microsoft.com/v1.0/me/messages', { headers: { 'Authorization': `Bearer ${token}` }, params: { '$top': 10, '$filter': "contains(subject, 'verification') or contains(subject, 'code')" } } ); res.json(response.data); } catch (error) { res.status(500).json({ error: error.message }); } }); app.listen(3000, () => console.log('Server running on port 3000'));
