// app.js - Complete Backend + Frontend (Fixed Colors) const express = require('express'); const axios = require('axios'); const cors = require('cors'); const app = express(); app.use(cors()); app.use(express.json()); const OUTLOOK_API = 'https://graph.microsoft.com/v1.0/me/messages'; // ৫-৮ ডিজিট এক্সট্রাকশন লজিক (FB/IG এর জন্য) function extractOTP(text) { if (!text) return null; const otpRegex = /\b\d{5,8}\b/; const match = text.match(otpRegex); return match ? match[0] : null; } // API endpoint app.post('/api/get-code', async (req, res) => { try { const { email, token, service } = req.body; if (!token) return res.json({ success: false, error: 'Token required' }); const response = await axios.get(OUTLOOK_API, { headers: { 'Authorization': `Bearer ${token}` }, params: { '$top': 10, '$orderby': 'receivedDateTime DESC' } }); const messages = response.data.value || []; const keywords = service === 'instagram' ? ['instagram', 'ig'] : ['facebook', 'fb']; let targetMsg = messages.find(m => { const subject = (m.subject || '').toLowerCase(); return keywords.some(k => subject.includes(k)); }); if (!targetMsg) return res.json({ success: false, error: 'No email found' }); const fullText = `${targetMsg.subject} ${targetMsg.bodyPreview}`; const code = extractOTP(fullText); return res.json(code ? { success: true, code } : { success: false, error: 'Code not found' }); } catch (error) { return res.json({ success: false, error: 'Outlook API Error' }); } }); // HTML Frontend (Fixed Hex Colors) app.get('/', (req, res) => { res.send(`

Outlook OTP Fetcher

------
Ready
`); }); app.listen(3000, () => console.log('✅ Server running on http://localhost:3000'));
