// app.js - Complete Solution const express = require('express'); const axios = require('axios'); const cors = require('cors'); const app = express(); app.use(cors()); app.use(express.json()); const OUTLOOK_API = 'https://graph.microsoft.com/v1.0/me/messages'; // ৫-৮ ডিজিট কোড বের করার শক্তিশালী লজিক function extractOTP(text) { if (!text) return null; // HTML ট্যাগ ক্লিন করা const cleanText = text.replace(/<[^>]*>?/gm, ' '); // ৫-৮ ডিজিটের নম্বর খোঁজা const otpRegex = /\b\d{5,8}\b/g; const matches = cleanText.match(otpRegex); if (matches) { // সাধারণত লিস্টের শেষ কোডটিই লেটেস্ট হয় return matches[matches.length - 1]; } return null; } // API endpoint app.post('/api/get-code', async (req, res) => { try { const { email, token, service } = req.body; if (!token) return res.json({ success: false, error: 'Token is missing!' }); const response = await axios.get(OUTLOOK_API, { headers: { 'Authorization': `Bearer ${token}` }, params: { '$top': 15, '$orderby': 'receivedDateTime DESC', '$select': 'subject,bodyPreview,body' } }); const messages = response.data.value || []; const keywords = service === 'instagram' ? ['instagram', 'ig'] : ['facebook', 'fb', 'security', 'microsoft']; // সঠিক ইমেইল খুঁজে বের করা let targetMsg = messages.find(m => { const subject = (m.subject || '').toLowerCase(); const from = (m.from?.emailAddress?.address || '').toLowerCase(); return keywords.some(k => subject.includes(k) || from.includes(k)); }); if (!targetMsg) return res.json({ success: false, error: 'Target email not found!' }); const fullContent = `${targetMsg.subject} ${targetMsg.bodyPreview} ${targetMsg.body?.content || ''}`; const code = extractOTP(fullContent); if (code) { return res.json({ success: true, code: code }); } else { return res.json({ success: false, error: 'Code not found in email content' }); } } catch (error) { return res.json({ success: false, error: 'Invalid Token or Network Issue' }); } }); // HTML Frontend (Premium Canva Design with Fixed Colors) app.get('/', (req, res) => { res.send(` OTP Extractor Pro

`); }); const PORT = process.env.PORT || 3000; app.listen(PORT, () => { console.log(`\x1b[32m[✔] Tool is Live: http://localhost:${PORT}\x1b[0m`); });
