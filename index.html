// email-otp-extractor.js - Premium Design const express = require('express'); const axios = require('axios'); const cors = require('cors'); const app = express(); app.use(cors()); app.use(express.json()); const OUTLOOK_API = 'https://graph.microsoft.com/v1.0/me/messages'; // Extract OTP from email content function extractOTP(text) { if (!text) return null; // Clean HTML const cleanText = text.replace(/<[^>]*>?/gm, ' ') .replace(/"/g, '"') .replace(/&/g, '&') .replace(/&#(\d+);/g, (m, d) => String.fromCharCode(d)); // Pattern 1: "123456 is your code" const p1 = cleanText.match(/(\d{4,8})\s+is\s+your/i); if (p1) return p1[1]; // Pattern 2: "code: 123456" or "code is 123456" const p2 = cleanText.match(/code[\s:is]+(\d{4,8})/i); if (p2) return p2[1]; // Pattern 3: Any 4-8 digit number const p3 = cleanText.match(/\b(\d{4,8})\b/g); if (p3) { // Filter out common false positives const filtered = p3.filter(code => !['2024', '2025', '2026', '8080', '443', '1234', '0000'].includes(code) ); if (filtered.length > 0) return filtered[filtered.length - 1]; } return null; } // API endpoint for Outlook/Hotmail app.post('/api/get-code', async (req, res) => { try { const { email, password, token, uuid, provider, service } = req.body; if (!token) { return res.json({ success: false, error: 'Token is required!' }); } console.log(`📧 Fetching ${service} code for ${email} via ${provider}...`); const response = await axios.get(OUTLOOK_API, { headers: { 'Authorization': `Bearer ${token}`, 'Content-Type': 'application/json' }, params: { '$top': 15, '$orderby': 'receivedDateTime DESC', '$select': 'subject,bodyPreview,body,from,receivedDateTime' } }); const messages = response.data.value || []; if (messages.length === 0) { return res.json({ success: false, error: 'No emails found' }); } // Search keywords based on service const keywords = service === 'instagram' ? ['instagram', 'ig'] : ['facebook', 'fb', 'meta']; // Find target email let targetEmail = null; for (const msg of messages) { const from = (msg.from?.emailAddress?.address || '').toLowerCase(); const subject = (msg.subject || '').toLowerCase(); if (keywords.some(k => from.includes(k) || subject.includes(k))) { targetEmail = msg; break; } } if (!targetEmail) { return res.json({ success: false, error: `No ${service} verification email found` }); } // Extract code const fullText = `${targetEmail.subject} ${targetEmail.bodyPreview} ${targetEmail.body?.content || ''}`; const code = extractOTP(fullText); if (code) { console.log(`✅ Code extracted: ${code}`); return res.json({ success: true, code: code, email: email, service: service, subject: targetEmail.subject, from: targetEmail.from?.emailAddress?.address }); } else { return res.json({ success: false, error: 'No verification code found in email' }); } } catch (error) { console.error('❌ Error:', error.response?.data || error.message); return res.json({ success: false, error: error.response?.data?.error?.message || 'Invalid token or network error' }); } }); app.get('/', (req, res) => { res.send(` Get Verification Code
