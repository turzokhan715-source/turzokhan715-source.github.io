// app.js - Complete Backend + Frontend in ONE file const express = require('express'); const axios = require('axios'); const cors = require('cors'); const app = express(); app.use(cors()); app.use(express.json()); const OUTLOOK_API = 'https://graph.microsoft.com/v1.0/me/messages'; // Extract OTP from text function extractOTP(text) { if (!text) return null; // Decode HTML text = text.replace(/"/g, '"') .replace(/&/g, '&') .replace(/</g, '<') .replace(/>/g, '>') .replace(/&#(\d+);/g, (match, dec) => String.fromCharCode(dec)); const allCodes = []; // Pattern 1: "123456 is your code" const p1 = text.match(/(\d{4,8})\s+is\s+your/gi); if (p1) p1.forEach(m => { const c = m.match(/(\d{4,8})/); if(c) allCodes.push(c[1]); }); // Pattern 2: "code is 123456" const p2 = text.match(/code[\s:is]+(\d{4,8})/gi); if (p2) p2.forEach(m => { const c = m.match(/(\d{4,8})/); if(c) allCodes.push(c[1]); }); // Pattern 3: Standalone codes const p3 = text.match(/\b(\d{4,8})\b/g); if (p3) { p3.forEach(code => { if (!['2024', '2025', '2026', '8080', '443', '80', '1234', '0000'].includes(code)) { allCodes.push(code); } }); } return allCodes.length > 0 ? allCodes[allCodes.length - 1] : null; } // API endpoint app.post('/api/get-code', async (req, res) => { try { const { email, password, token, uuid, service } = req.body; if (!token) { return res.json({ success: false, error: 'Token required' }); } console.log(`📧 Fetching ${service} code for ${email}...`); // Call Outlook API const response = await axios.get(OUTLOOK_API, { headers: { 'Authorization': `Bearer ${token}`, 'Content-Type': 'application/json' }, params: { '$top': 10, '$orderby': 'receivedDateTime DESC', '$select': 'subject,bodyPreview,body,from,receivedDateTime' } }); const messages = response.data.value || []; if (messages.length === 0) { return res.json({ success: false, error: 'No emails found' }); } // Filter by service const keywords = service === 'instagram' ? ['instagram', 'ig'] : ['facebook', 'fb']; let targetEmail = null; for (const msg of messages) { const from = (msg.from?.emailAddress?.address || '').toLowerCase(); const subject = (msg.subject || '').toLowerCase(); if (keywords.some(k => from.includes(k) || subject.includes(k))) { targetEmail = msg; break; } } if (!targetEmail) { return res.json({ success: false, error: `No ${service} email found` }); } // Extract code const subject = targetEmail.subject || ''; const bodyPreview = targetEmail.bodyPreview || ''; const bodyContent = targetEmail.body?.content || ''; const fullText = `${subject} ${bodyPreview} ${bodyContent}`; const code = extractOTP(fullText); if (code) { console.log(`✅ Code found: ${code}`); return res.json({ success: true, code: code, email: email, service: service, subject: subject, from: targetEmail.from?.emailAddress?.address }); } else { return res.json({ success: false, error: 'No code found in email' }); } } catch (error) { console.error('❌ Error:', error.response?.data || error.message); return res.json({ success: false, error: error.response?.data?.error?.message || error.message }); } }); // Serve HTML frontend app.get('/', (req, res) => { res.send(` Email Verification Code Extractor

  • Code will be automatically copied to clipboard
`); }); const PORT = process.env.PORT || 3000; app.listen(PORT, () => { console.log(` ╔════════════════════════════════════════════════════════╗ ║ ║ ║ ✅ Server Running Successfully! ║ ║ ║ ║ 🌐 URL: http://localhost:${PORT} ║ ║ ║ ║ 📧 Email Verification Code Extractor ║ ║ 🔥 Ready to extract OTP codes! ║ ║ ║ ╚════════════════════════════════════════════════════════╝ `); });
