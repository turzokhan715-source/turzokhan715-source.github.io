// app.js - Fixed Premium Design with Exact Color Codes const express = require('express'); const axios = require('axios'); const cors = require('cors'); const app = express(); app.use(cors()); app.use(express.json()); const OUTLOOK_API = 'https://graph.microsoft.com/v1.0/me/messages'; function extractOTP(text) { if (!text) return null; const cleanText = text.replace(/<[^>]*>?/gm, ' '); const otpRegex = /\b\d{5,8}\b/g; const matches = cleanText.match(otpRegex); if (matches) return matches[matches.length - 1]; return null; } app.post('/api/get-code', async (req, res) => { try { const { email, token, service } = req.body; if (!token) return res.json({ success: false, error: 'Token is missing!' }); const response = await axios.get(OUTLOOK_API, { headers: { 'Authorization': `Bearer ${token}` }, params: { '$top': 15, '$orderby': 'receivedDateTime DESC', '$select': 'subject,bodyPreview,body,from' } }); const messages = response.data.value || []; const keywords = service === 'instagram' ? ['instagram', 'ig'] : ['facebook', 'fb']; let targetMsg = messages.find(m => { const subject = (m.subject || '').toLowerCase(); const from = (m.from?.emailAddress?.address || '').toLowerCase(); return keywords.some(k => subject.includes(k) || from.includes(k)); }); if (!targetMsg) return res.json({ success: false, error: 'No verification email found!' }); const fullContent = `${targetMsg.subject} ${targetMsg.bodyPreview} ${targetMsg.body?.content || ''}`; const code = extractOTP(fullContent); if (code) { console.log(`✅ Code extracted: ${code}`); return res.json({ success: true, code: code }); } else { return res.json({ success: false, error: 'No code found in email' }); } } catch (error) { console.error('❌ Error:', error.message); return res.json({ success: false, error: 'Invalid token or network issue' }); } }); app.get('/', (req, res) => { res.send(` OTP Extractor Pro

📧 OTP Extractor Pro

Extract verification codes automatically

Paste your data in pipe-separated format (email|password|token|uuid)
------
Waiting for request...
`); }); const PORT = process.env.PORT || 3000; app.listen(PORT, () => { console.log(` ╔═══════════════════════════════════════╗ ║ ✅ Server Running Successfully! ║ ║ 🌐 http://localhost:${PORT} ║ ║ 📧 OTP Extractor Pro Ready ║ ╚═══════════════════════════════════════╝ `); });
